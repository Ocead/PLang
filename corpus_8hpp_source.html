21 // Code from boost
22 // Reciprocal of the golden ratio helps spread entropy
23 // and handles duplicates.
24 // See Mike Seymour in magic-numbers-in-boosthash-combine:
25 // http://stackoverflow.com/questions/4948780
26 template<class T>
27 inline constexpr void hash_combine(std::size_t &seed, T const &v) {
28 seed ^= std::hash<T>()(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
29 }
30
31 // Recursive template code derived from Matthieu M.
32 template<class Tuple, size_t Index = std::tuple_size<Tuple>::value - 1>
33 struct hash_value_impl {
34 static constexpr void apply(size_t &seed, Tuple const &tuple) {
35 hash_value_impl<Tuple, Index - 1>::apply(seed, tuple);
36 hash_combine(seed, std::get<Index>(tuple));
37 }
38 };
39
40 template<class Tuple>
41 struct hash_value_impl<Tuple, 0> {
42 static constexpr void apply(size_t &seed, Tuple const &tuple) {
43 hash_combine(seed, std::get<0>(tuple));
44 }
45 };
46 }// namespace

48 template<typename T>
49 struct hash<plang::detail::corpus::tag<T>> {
50 constexpr size_t operator()(plang::detail::corpus::tag<T> const &) const {
51 return typeid(T).hash_code();
52 }

55 template<typename... Tps>
56 struct hash<std::tuple<Tps...>> {
57 constexpr size_t operator()(std::tuple<Tps...> const &tt) const {
58 size_t seed = 0;
59 hash_value_impl<std::tuple<Tps...>>::apply(seed, tt);
60 return seed;
61 }

158 public:
159 using Base = std::tuple<entry, std::vector<entry>, action>;
160 using Base::Base;
161
162 template<typename T>
163 resolve_entry_result(resolve_result<T> &&result)
164 : Base(result.has_result() ? entry(result.entry()) : entry(), {}, result.action()) {
165 auto &can = candidates();
166 can.reserve(result.candidates().size());
167 std::transform(result.candidates().begin(),
168 result.candidates().end(),
169 std::make_move_iterator(can.begin()),
170 [](T &&r) -> plang::entry { return std::forward<T>(r); });
171 }
