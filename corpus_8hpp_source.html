22 // Code from boost
23 // Reciprocal of the golden ratio helps spread entropy
24 // and handles duplicates.
25 // See Mike Seymour in magic-numbers-in-boosthash-combine:
26 // http://stackoverflow.com/questions/4948780
27 template<class T>
28 inline constexpr void hash_combine(std::size_t &seed, T const &v) {
29 seed ^= std::hash<T>()(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
30 }
31
32 // Recursive template code derived from Matthieu M.
33 template<class Tuple, size_t Index = std::tuple_size<Tuple>::value - 1>
34 struct hash_value_impl {
35 static constexpr void apply(size_t &seed, Tuple const &tuple) {
36 hash_value_impl<Tuple, Index - 1>::apply(seed, tuple);
37 hash_combine(seed, std::get<Index>(tuple));
38 }
39 };
40
41 template<class Tuple>
42 struct hash_value_impl<Tuple, 0> {
43 static constexpr void apply(size_t &seed, Tuple const &tuple) {
44 hash_combine(seed, std::get<0>(tuple));
45 }
46 };
47 }// namespace

56 template<typename... Tps>
57 struct hash<std::tuple<Tps...>> {
58 constexpr size_t operator()(std::tuple<Tps...> const &tt) const {
59 size_t seed = 0;
60 hash_value_impl<std::tuple<Tps...>>::apply(seed, tt);
61 return seed;
62 }

152 class resolve_entry_result : public std::tuple<entry, std::vector<entry>, action> {
153 public:
154 using Base = std::tuple<entry, std::vector<entry>, action>;
155 using Base::Base;
156
157 template<typename T>
158 resolve_entry_result(resolve_result<T> &&result)
159 : Base(result.has_result() ? entry(result.entry()) : entry(), {}, result.action()) {
160 auto &can = candidates();
161 can.reserve(result.candidates().size());
162 std::transform(result.candidates().begin(),
163 result.candidates().end(),
164 std::make_move_iterator(can.begin()),
165 [](T &&r) -> plang::entry { return std::forward<T>(r); });
166 }
